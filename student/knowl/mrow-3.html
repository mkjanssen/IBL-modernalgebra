\begin{align*} 1^3 + 2^3 + 3^3 + \cdots + n^3 +(n+1)^3 \amp = \frac{n^2(n+1)^2}{4} + (n+1)^3 \\ \amp = \frac{n^2(n+1)^2}{4} + \frac{4(n+1)^3}{4}\\ \amp = \frac{n^2(n+1)^2 + 4(n+1)^3}{4}\\ \amp = \frac{(n+1)^2 (n^2+4(n+1))}{4}\\ \amp = \frac{(n+1)^2 (n^2+4n+4)}{4}\\ \amp = \frac{(n+1)^2 (n+2)^2}{4}. \end{align*}
